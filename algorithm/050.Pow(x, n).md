## 50. Pow(x, n) [E]
Implement pow(x, n).

# Code:
- Solution 1 - stupid way (20 ms):
```c++
class Solution 
{
public:
    double myPow(double x, int n) 
    {
        if (x == 1)
            return 1;
        if (x == -1)
            return n % 2 == 1 ? -1 : 1;
            
        if ((n == 2147483647) || (n == -2147483648))
            return 0;
            
        double ans = 1;
        if (n > 0)
            for (int i = 0; i < n; i ++)
                ans *= x;
        else if (n == 0)
            return 1;
        else
            for (int i = 0; i > n; i --)
                ans *= (1 / x);
        
        return x == 0 ? 0 : ans;
    }
};
```
