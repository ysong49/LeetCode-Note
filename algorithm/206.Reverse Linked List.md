## 206. Reverse Linked List [E]
Reverse a singly linked list.

## Code:
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution 
{
public:
    ListNode* reverseList(ListNode* head) 
    {
        if (!head || !head -> next)
            return head;
        
        ListNode* prev = head, *curr = head -> next, * next;
        
        while (curr)
        {
            next = curr -> next;
            curr -> next = prev;
            
            prev = curr;
            curr = next;
        }
        
        head -> next = NULL;
        return prev;
    }
};
```
