## 404. Sum of Left Leaves [E]
Find the sum of all left leaves in a given binary tree.

#### Example:
```
    3
   / \
  9  20
    /  \
   15   7

There are two left leaves in the binary tree, with values 9 and 15 respectively. Return 24.
```

## Code
### 2 Round
- Solution 1 - recursion (3 ms):
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    // 2nd round        date: 2016-10-09        location: Vista Del Lago III Apartement
public:
    int sumOfLeftLeaves(TreeNode* root) {
        int res = 0;
        helper(root, res, false);
        return res;
    }
    
    void helper(TreeNode* node, int& res, bool isLeft) {
        if (!node)  return;
        if (isLeft && !node -> left && !node -> right)
            res += node -> val;
        helper(node -> left, res, true);
        helper(node -> right, res, false);
    }
};
```

- Solution 2 - iteration (6 ms):
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    // 2nd round        date: 2016-10-09        location: Vista Del Lago III Apartement
public:
    int sumOfLeftLeaves(TreeNode* root) {
        if (!root)  return 0;
        stack<bool> isLeft;
        stack<TreeNode*> nodes;
        int res = 0;
        nodes.push(root);
        isLeft.push(false);
        while (!nodes.empty()) {
            TreeNode* node = nodes.top(); 
            nodes.pop();
            if (isLeft.top() && !node->left && !node->right)
                res += node->val;
            isLeft.pop();
            if (node->left) {
                nodes.push(node->left);
                isLeft.push(true);
            }
            if (node->right) {
                nodes.push(node->right);
                isLeft.push(false);
            }
        }
        return res;
    }
};
```
