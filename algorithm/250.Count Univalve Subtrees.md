## 250. Count Univalue Subtrees [M]
Given a binary tree, count the number of uni-value subtrees.

A Uni-value subtree means all nodes of the subtree have the same value.

For example:
Given binary tree,
```
              5
             / \
            1   5
           / \   \
          5   5   5
```
return `4`.

## Code:
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution 
{
    // date: 2016-08-14     location: Santa Clara City Library
public:
    int countUnivalSubtrees(TreeNode* root) 
    {
        int res = 0;
        helper(root, res);
        return res;
    }
    
    bool helper(TreeNode* node, int& res)
    {
        if (!node)  return true;
        bool left = helper(node -> left, res), right = helper(node -> right, res);
        
        if (left && right 
         && ((node -> left == NULL || node -> left -> val == node -> val) 
         && ( node -> right == NULL || node -> right -> val == node -> val)))
        {
            res ++;
            return true;
        }
        return false;
    }
};
```
