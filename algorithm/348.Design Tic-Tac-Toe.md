## 348. Design Tic-Tac-Toe [M]
Design a Tic-tac-toe game that is played between two players on a n x n grid.

You may assume the following rules:

1. A move is guaranteed to be valid and is placed on an empty block.
2. Once a winning condition is reached, no more moves is allowed.
3. A player who succeeds in placing n of their marks in a horizontal, vertical, or diagonal row wins the game.

##### Example:
```
Given n = 3, assume that player 1 is "X" and player 2 is "O" in the board.

TicTacToe toe = new TicTacToe(3);

toe.move(0, 0, 1); -> Returns 0 (no one wins)
|X| | |

| | | |

toe.move(0, 2, 2); -> Returns 0 (no one wins)
|X| |O|
| | | |    // Player 2 makes a move at (0, 2).
| | | |

toe.move(2, 2, 1); -> Returns 0 (no one wins)
|X| |O|
| | | |    // Player 1 makes a move at (2, 2).
| | |X|

toe.move(1, 1, 2); -> Returns 0 (no one wins)
|X| |O|
| |O| |    // Player 2 makes a move at (1, 1).
| | |X|

toe.move(2, 0, 1); -> Returns 0 (no one wins)
|X| |O|
| |O| |    // Player 1 makes a move at (2, 0).
|X| |X|

toe.move(1, 0, 2); -> Returns 0 (no one wins)
|X| |O|
|O|O| |    // Player 2 makes a move at (1, 0).
|X| |X|

toe.move(2, 1, 1); -> Returns 1 (player 1 wins)
|X| |O|
|O|O| |    // Player 1 makes a move at (2, 1).
|X|X|X|
```

### Code:
```c++
class TicTacToe 
{
    // date: 2016-07-31     location: Coffee Factory
    vector<vector<int>> board;
public:
    /** Initialize your data structure here. */
    TicTacToe(int n) 
    {
        vector<int> temp(n, 0);
        for (int i = 0; i < n; i ++)
            board.push_back(temp);
    }
    
    /** Player {player} makes a move at ({row}, {col}).
        @param row The row of the board.
        @param col The column of the board.
        @param player The player, can be either 1 or 2.
        @return The current winning condition, can be either:
                0: No one wins.
                1: Player 1 wins.
                2: Player 2 wins. */
    int move(int row, int col, int player) 
    {
        board[row][col] = player;
        bool horizontal = true, vertical = true, diagonal = true, anti_diagonal = true;
        for (int i = 0; i < board.size(); i ++)
            if (board[row][i] != player)
            {
                vertical = false;
                break;
            }
        if (vertical)   return player;
        
        for (int i = 0; i < board.size(); i ++)
            if (board[i][col] != player)
            {
                horizontal = false;
                break;
            }
        if (horizontal)   return player;
        
        if (row == col)
        {
            for (int i = 0; i < board.size(); i ++)
                if (board[i][i] != player)
                {
                    diagonal = false;
                    break;
                }
        }
        else
            diagonal = false;
        if (diagonal)   return player;
   
        if (board.size() - 1 == col + row)
        {
            for (int i = 0; i < board.size(); i ++)
                if (board[i][board.size() - 1 - i] != player)
                {
                    anti_diagonal = false;
                    break;
                }
        }
        else
            anti_diagonal = false;
        if (anti_diagonal)   return player;
 
        //return (horizontal || vertical || diagonal || anti_diagonal) ? player : 0;
        return false;
    }
};
 ```
